public static void showTravel(Date outbound, Date inbound, String origin, String destination) {
	Airfare airfare = Airfare.findFor(origin, destination, outbound, inbound);
		if(airfare != null) {
			Hotel hotel = Hotel.findHotelFor(origin, destination, outbound, inbound);
			if(hotel != null) { 
				if(airfare.arrivalHour!= null && airfare.arrivalHour > 23) { 
					Car car = Car.findCarFor(Date outbound, airfare.arrivalAirport, airfare.arrivalHour);
					render(airfare, hotel, car);           
				} else {
					render(airfare, hotel, null); // la vue se débrouillera avec le null...               
				}
			} else {
				renderTemplate("Application/noHotel.html");
			}    
		} else { 
			renderTemplate("Application/noAirfare.html");
		}      
	}
